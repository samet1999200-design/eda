<script>
function patlat() {
    document.getElementById("mesaj").innerHTML =
    "Seni seviyorum 💖 İyi ki hayatımdasın 🫶";

    for(let i=0; i<25; i++) {
        efektOlustur();
    }
}

function efektOlustur() {
    const element = document.createElement("div");
    element.classList.add("effect");

    const emojis = ["💖","💗","💘","🧸"];
    element.innerHTML = emojis[Math.floor(Math.random() * emojis.length)];

    element.style.left = Math.random() * 100 + "vw";
    element.style.top = "60%";

    document.body.appendChild(element);

    setTimeout(() => {
        element.remove();
    }, 4000);
}
